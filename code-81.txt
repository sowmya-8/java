Given a positive integer, check if it is a palindrome. If it is a palindrome, output the count of its non-zero digits; otherwise, output 0

Input Format

A single positive integer N

Constraints

1 ≤ N ≤ 109

Output Format

Output the count of non-zero digits if N is a palindrome; otherwise, output 0.

Sample Input 0

1221
Sample Output 0

4
Sample Input 1

1234
Sample Output 1

0
---------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------------
import java.util.Scanner;

public class PalindromeDigitCounter {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String N = scanner.nextLine().trim();
        scanner.close();

        boolean isPalindrome = true;
        int left = 0, right = N.length() - 1;
        while (left < right) {
            if (N.charAt(left) != N.charAt(right)) {
                isPalindrome = false;
                break;
            }
            left++;
            right--;
        }

        if (isPalindrome) {
            int count = 0;
            for (char c : N.toCharArray()) {
                if (c != '0') {
                    count++;
                }
            }
            System.out.println(count);
        } else {
            System.out.println(0);
        }
    }
}
