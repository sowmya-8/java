import java.util.Scanner;

public class Solution {
    public static void main(String[] args) {
        Scanner scanner = new Scanner(System.in);
        String input = scanner.nextLine();

        int score = 0;
        int multiplier = 1;

        for (int i = input.length() - 1; i >= 0; i--) {
            int position = input.charAt(i) - 'a' + 1;
            score += position * multiplier;
            multiplier = -(multiplier + (multiplier > 0 ? 1 : -1));
        }

        int finalScore = Math.abs(score);

        if (finalScore <= 50) {
            System.out.println("Balanced Word");
        } else if (finalScore <= 100) {
            System.out.println("Risky Word");
        } else {
            System.out.println("Chaotic Word");
        }
    }
}
